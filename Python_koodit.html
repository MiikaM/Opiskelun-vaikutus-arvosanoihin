from datetime import datetime

data['Paiva'] = pd.to_datetime(data['Paiva'], format="%d.%m.%Y %H:%M")
df=data
df["Kuukausi"] = df['Paiva'].dt.to_period('M')
kk = df.groupby('Pisteet')['Paiva'].value_counts()
min = df.groupby('Pisteet')['Paiva'].min()
max = df.groupby('Pisteet')['Paiva'].max()

max = max.to_frame()
min = min.to_frame()
timeframe = (max-min)

timeframe['pisteet_kurssilla'] = list(timeframe.index)
timeframe['aloitus'] = min
timeframe['lopetus'] = max

timeframe.rename(columns={'Paiva':'toimintojen_aikavali'}, inplace=True)
timeframe['toimintojen_aikavali_paivina'] = list(timeframe.toimintojen_aikavali.dt.total_seconds() / (60 * 60 * 24))

fig=plt.figure(figsize=(18, 10), dpi= 80, facecolor='w', edgecolor='k')

plt.plot(timeframe.aloitus, timeframe.pisteet_kurssilla, '*', label='aloitus_paiva')
plt.plot(timeframe.lopetus, timeframe.pisteet_kurssilla, '.',  color="red", label='lopetus_paiva')
plt.ylabel('pisteet_kurssilla')
plt.legend()
plt.savefig('aloitus_ja_lopetus.png')

timeframe.plot(kind='scatter',x='toimintojen_aikavali_paivina',y='pisteet_kurssilla',color='red')
plt.savefig('kesto_paivina.png')
